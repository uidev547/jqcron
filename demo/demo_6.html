$(function(){
    var cron =
        $('.example5')
        .jqCron()
        .jqCronGetInstance();
     
    $('.a5-enable').click(function(e){
        cron.enable();
        e.preventDefault();
    });
     
    $('.a5-disable').click(function(e){
        cron.disable();
        e.preventDefault();
    });
     
    $('.a5-toggle').click(function(e){
        if(cron.isDisabled())
            cron.enable();
        else
            cron.disable();
        e.preventDefault();
    });
});
